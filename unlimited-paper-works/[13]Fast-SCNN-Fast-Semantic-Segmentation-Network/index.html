class SeparableConv2d(nn.Module):
    def __init__(self, inplanes, planes, kernel_size=3, stride=1, dilation=1, relu_first=True,
                 bias=False, norm_layer=nn.BatchNorm2d):
        super().__init__()
        # æ·±åº¦å·ç§¯ï¼Œå·ç§¯æ ¸ä¸º3æ­¥é•¿1ï¼Œpadding1ï¼Œç©ºæ´1çš„å·ç§¯å±‚
        depthwise = nn.Conv2d(inplanes, inplanes, kernel_size,
                              stride=stride, padding=dilation,
                              dilation=dilation, groups=inplanes, bias=bias)
        # å¯¹åº”çš„BN

        # ç‚¹å·ç§¯ï¼Œå°±æ˜¯æ™®é€šçš„1x1å·ç§¯
        pointwise = nn.Conv2d(inplanes, planes, 1, bias=bias)
        # å¯¹åº”çš„BN

        # æ˜¯å¦ä½¿ç”¨æ¿€æ´»å‡½æ•°

                                                    ('depthwise', depthwise),

                                                    ('pointwise', pointwise),
                                                    ('bn_point', bn_point)
                                                    ]))
        else:
            self.block = nn.Sequential(OrderedDict([('depthwise', depthwise),
                                                    ('bn_depth', bn_depth),
                                                    ('relu1', nn.ReLU(inplace=True)),
                                                    ('pointwise', pointwise),
                                                    ('bn_point', bn_point),
                                                    ('relu2', nn.ReLU(inplace=True))
                                                    ]))

    def forward(self, x):
        return self.block(x)

class InvertedResidual(nn.Module):
    def __init__(self, in_channels, out_channels, stride, expand_ratio, dilation=1, norm_layer=nn.BatchNorm2d):
        # å‚æ•°ï¼šè¾“å…¥é€šé“ï¼Œè¾“å‡ºé€šé“ï¼Œæ­¥é•¿ï¼Œæ‹“å±•å› å­ï¼Œç©ºæ´å·ç§¯ï¼Œ
        super(InvertedResidual, self).__init__()
        assert stride in [1, 2]
        # æ˜¯å¦ä½¿ç”¨æ®‹å·®è¿æ¥
        self.use_res_connect = stride == 1 and in_channels == out_channels

        layers = list()
        # ä¸­é—´çš„é€šé“æ•°ï¼Œä½¿ç”¨æ‹“å±•å› å­*è¾“å…¥çš„é€šé“æ•°
        inter_channels = int(round(in_channels * expand_ratio))
        if expand_ratio != 1:
            # pw
            # å…ˆåšä¸€ä¸ªæ ‡å‡†å·ç§¯å˜›ï¼Œä½¿ç”¨ä¸­é—´é€šé“æ•°ä½œä¸ºè¾“å‡º,1x1å·ç§¯
            layers.append(_ConvBNReLU(in_channels, inter_channels, 1, relu6=True, norm_layer=norm_layer))
        layers.extend([
            # dw è¿™é‡Œä½¿ç”¨äº†åˆ†ç»„å·ç§¯ï¼Œä½†æ˜¯å®é™…ä¸Šå’Œæ™®é€šçš„å·ç§¯æ²¡æœ‰ä»€ä¹ˆåŒºåˆ«ï¼Œå¦‚æœgroupsæ•´å¥½æ˜¯è¾“å…¥é€šé“æ•°çš„ä¸€ä¸ªå› ç´ ï¼Œåˆ™è¾“å…¥çš„é€šé“ä¼šè¢«åˆ†æˆå¯¹åº”çš„ç»„è¿›è¡Œå·ç§¯

                        groups=inter_channels, relu6=True, norm_layer=norm_layer),
            # pw-linear
            # ä½¿ç”¨1x1å·ç§¯å°†ä¸­é—´é€šé“æ•°è½¬åŒ–æˆæœ€ç»ˆçš„é€šé“æ•°

            norm_layer(out_channels)])
        self.conv = nn.Sequential(*layers)

    def forward(self, x):
        # æ®‹å·®è¿æ¥
        if self.use_res_connect:
            return x + self.conv(x)
        else:
            return self.conv(x)

class PyramidPooling(nn.Module):
